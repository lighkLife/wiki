# 数据权限

```uml
enum DpOperationEnum {
    READ
    WRITE
}

interface IdMapping {
    + mapToAtomicResources(List<Long> resourceIds): Set<DpResourceBo>
}

class AbstractDpHandler implements IdMapping {
    + validate(DpOperationEnum operationEnum, List<Long> resourceIds)
}



class TradingRetailNoteHandler extends AbstractDpHandler {
    - tradingRetailHandler: TradingRetailHandler
    - operationLogMapper: OperationLogMapper
    + mapToAtomicResources(List<Long> resourceIds): Set<DpResourceBo>
}

class TradingRetailHandler extends AbstractDpHandler {
    - retailStandingBookMapper: RetailStandingBookMapper
    + mapToAtomicResources(List<Long> resourceIds): Set<DpResourceBo>
}

class TradingRetailFileHandler extends AbstractDpHandler {
    - tradingRetailHandler: TradingRetailHandler
    - enclosureMapper: EnclosureMapper
    + mapToAtomicResources(List<Long> resourceIds): Set<DpResourceBo>
}

TradingRetailNoteHandler::tradingRetailHandler --> TradingRetailHandler
TradingRetailFileHandler::tradingRetailHandler --> TradingRetailHandler
```
